merged_df <- merged_df %>%
    mutate(
        aggr_spc = 1 + str_count(species_scientific_name, "/"),
        genus = word(species_scientific_name, 1),
        genus_code = substr(species_abbreviation, 1, 3),
        year  = year(date),

        lat_bin = floor(latitude / 10) * 10,

        drop_na()
