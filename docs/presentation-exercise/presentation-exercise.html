reactable(
  table_chicken,
  columns = list(
    region = colDef(name = "Country"),
    
    latest_2023 = colDef(
      name = "2023 Production (t)",
      format = colFormat(digits = 0),
      defaultSortOrder = "desc"
    ),
    
    change = colDef(
      name = "2023 change (%)",
      format = colFormat(digits = 1),
      style = function(value) {
        color <- if (value > 5) "green"
        else if (value > -5) "black"
        else "red"
        list(
          color = color,
          fontWeight = "bold"
        )
      },
    ),
    
    trend = colDef(
      name = "Historic Trend",
      cell = function(values) {
        sparkline(values, type = "line")
      },
      html = TRUE
    )
  ),
  searchable = TRUE,
  sortable = TRUE,
  highlight = TRUE,
  striped = TRUE
)
